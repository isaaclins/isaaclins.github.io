{{- /* Image render hook for markdown ![alt](src) images. - Adds loading="lazy" + decoding="async" to below-the-fold images; the first image on the page (Ordinal 0, likely the LCP element) stays eager with high fetch priority. - When the source is a local /images/*.{png,jpg,jpeg} raster and a .webp sibling exists on disk (generated by .github/workflows/optimize-images.yml), wraps the in a that prefers the WebP. */ -}} {{- $dest := .Destination -}} {{- $alt := .PlainText -}} {{- $title := .Title -}} {{- $isFirst := eq .Ordinal 0 -}} {{- $loading := cond $isFirst "eager" "lazy" -}} {{- /* Parse the URL so a ?query (e.g. ?raw=true) doesn't break the sibling lookup. */ -}} {{- $path := (urls.Parse $dest).Path -}} {{- $ext := strings.ToLower (path.Ext $path) -}} {{- $isLocalImages := and (hasPrefix $path "/images/") (in (slice ".png" ".jpg" ".jpeg") $ext) -}} {{- $webpURL := "" -}} {{- if $isLocalImages -}} {{- $rel := strings.TrimPrefix "/images/" $path -}} {{- $base := strings.TrimSuffix $ext $rel -}} {{- $diskWebp := printf "images/%s.webp" $base -}} {{- if os.FileExists $diskWebp -}} {{- $webpURL = printf "/images/%s.webp" $base -}} {{- end -}} {{- end -}} {{- if $webpURL -}} {{- else -}} {{- end -}}
